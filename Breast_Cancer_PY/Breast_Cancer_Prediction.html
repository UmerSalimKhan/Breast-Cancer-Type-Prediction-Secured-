def optimalKNN ():
    acc_manhat = []
    acc_euc = []
    neigh_man = 0
    neigh_euc = 0
    
    for i in range (1, 3):
        for j in range (1, 31): 
            model = KNeighborsClassifier (n_neighbors=j, p=i, metric='minkowski') 
            model.fit (x_train, y_train)
            y_pred = model.predict (x_test)
            if i == 1:
                acc_manhat.append (accuracy_score (y_test, y_pred))
            elif i == 2:
                acc_euc.append (accuracy_score (y_test, y_pred))
                
    visual_optimal (acc_manhat, acc_euc)
    
    print ("Accuracy (Manhattan): ", acc_manhat)
    print ("\nAccuracy (Euclidean): ", acc_euc)

        if (acc_euc [i] == 0.9824561403508771):
            print ('Euclidean Neighbor = {}'.format (i))
